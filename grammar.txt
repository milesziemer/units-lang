expr    : "let" IDENTIFIER "=" [ arith ]
        : [ arith ]

arith   : [ rational ] { "+" | "-" } [ rational ]
        : [ rational ]

rational    : [ exp ] { "*" | "/" } [ exp ]
            : [ exp ]

exp         : [ apply ] "^" [ apply ]
            : [ apply ]

apply       : { "+" | "-" } [ unit ]
            : [ unit ]

unit        : NUMBER
            : IDENTIFIER
            : "(" [ expr ] ")"
